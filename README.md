package ui; import engine.StationEngine; import model.ColonyTask; import persistence.SaveLoadService; import resources.Resource; import resources.ResourceManager; import javax.swing.*; import java.awt.*; public class MainDashboard extends JFrame { private StationEngine engine; private JTextArea logArea; private DefaultListModel<String> queueModel; private JLabel partsLabel, oxygenLabel, rationsLabel, creditsLabel; private JComboBox<Resource> supplyDropdown; public MainDashboard(StationEngine engine) { this.engine = engine; setTitle("Ares Base Command - Team 32"); setSize(900, 600); setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); setLayout(new BorderLayout(10, 10)); engine.setCallbacks( message -> logMessage(message), v -> refreshUI() ); buildUI(); refreshUI(); engine.startEngine(); } private void buildUI() { JPanel resourcePanel = new JPanel(new FlowLayout(FlowLayout.CENTER, 20, 10)); resourcePanel.setBackground(Color.DARK_GRAY); partsLabel = createStyledLabel("Parts: 0"); oxygenLabel = createStyledLabel("Oxygen: 0"); rationsLabel = createStyledLabel("Rations: 0"); creditsLabel = createStyledLabel("Credits: 0"); creditsLabel.setForeground(Color.YELLOW); resourcePanel.add(partsLabel); resourcePanel.add(oxygenLabel); resourcePanel.add(rationsLabel); resourcePanel.add(creditsLabel); add(resourcePanel, BorderLayout.NORTH); logArea = new JTextArea(); logArea.setEditable(false); logArea.setBackground(Color.BLACK); logArea.setForeground(Color.GREEN); logArea.setFont(new Font("Monospaced", Font.PLAIN, 14)); JScrollPane logScroll = new JScrollPane(logArea); logScroll.setBorder(BorderFactory.createTitledBorder("System Log")); add(logScroll, BorderLayout.CENTER); queueModel = new DefaultListModel<>(); JList<String> queueList = new JList<>(queueModel); queueList.setBackground(new Color(40, 40, 40)); queueList.setForeground(Color.WHITE); JScrollPane queueScroll = new JScrollPane(queueList); queueScroll.setPreferredSize(new Dimension(300, 0)); queueScroll.setBorder(BorderFactory.createTitledBorder("Task Queue (Max 5)")); add(queueScroll, BorderLayout.EAST); JPanel bottomWrapper = new JPanel(new BorderLayout()); JPanel supplyPanel = new JPanel(new FlowLayout()); supplyPanel.setBorder(BorderFactory.createTitledBorder("Supply Chain (Cost: 100 Cr for 10 Units)")); Resource[] buyableItems = {Resource.SPARE_PARTS, Resource.OXYGEN, Resource.RATIONS}; supplyDropdown = new JComboBox<>(buyableItems); JButton buyBtn = new JButton("Buy Selected Supply"); buyBtn.addActionListener(e -> { Resource selected = (Resource) supplyDropdown.getSelectedItem(); engine.buySupplies(selected); }); supplyPanel.add(new JLabel("Select Resource: ")); supplyPanel.add(supplyDropdown); supplyPanel.add(buyBtn); JPanel controlPanel = new JPanel(new FlowLayout()); JButton execBtn = new JButton("Execute Top Task"); JButton saveBtn = new JButton("Save Game"); JButton loadBtn = new JButton("Load Game"); execBtn.addActionListener(e -> engine.executeNextTask()); saveBtn.addActionListener(e -> { try { SaveLoadService.saveGame(engine.getResourceManager()); logMessage("SYSTEM: Game Saved Successfully."); } catch (Exception ex) { logMessage("ERROR: " + ex.getMessage()); } }); loadBtn.addActionListener(e -> { try { SaveLoadService.loadGame(engine.getResourceManager()); refreshUI(); logMessage("SYSTEM: Game Loaded Successfully."); } catch (Exception ex) { logMessage("ERROR: Save file not found."); } }); controlPanel.add(execBtn); controlPanel.add(saveBtn); controlPanel.add(loadBtn); bottomWrapper.add(supplyPanel, BorderLayout.NORTH); bottomWrapper.add(controlPanel, BorderLayout.SOUTH); add(bottomWrapper, BorderLayout.SOUTH); } private JLabel createStyledLabel(String text) { JLabel label = new JLabel(text); label.setForeground(Color.WHITE); label.setFont(new Font("Arial", Font.BOLD, 16)); return label; } private void logMessage(String msg) { logArea.append("> " + msg + "\n"); logArea.setCaretPosition(logArea.getDocument().getLength()); } private void refreshUI() { ResourceManager rm = engine.getResourceManager(); partsLabel.setText("Spare Parts: " + rm.getAmount(Resource.SPARE_PARTS)); oxygenLabel.setText("Oxygen: " + rm.getAmount(Resource.OXYGEN)); rationsLabel.setText("Rations: " + rm.getAmount(Resource.RATIONS)); creditsLabel.setText("Credits: " + rm.getAmount(Resource.CREDITS)); queueModel.clear(); for (ColonyTask task : engine.getTaskQueue()) { // Displays: Task Name [TYPE] - Cost: X Parts queueModel.addElement(task.getName() + " [" + task.getTaskType() + "] Cost: " + task.getreqParts() + " Parts"); } } }
